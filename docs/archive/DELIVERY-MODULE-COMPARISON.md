## Database Schema (Inferred)

### delivery_schedule

```sql
CREATE TABLE delivery_schedule (
  id uuid PRIMARY KEY,
  order_id uuid REFERENCES orders(id),
  scheduled_date date NOT NULL,
  scheduled_time_window text,  -- "AM", "PM", "Morning", etc.
  status text DEFAULT 'scheduled',  -- 'scheduled', 'in_transit', 'delivered'
  actual_delivery_time timestamptz,
  driver_id uuid REFERENCES delivery_drivers(id),  -- If driver tracking
  vehicle_id uuid REFERENCES delivery_vehicles(id),  -- If vehicle tracking
  notes text,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);
```

### route_cache

```sql
CREATE TABLE route_cache (
  id uuid PRIMARY KEY,
  origin_location_id text NOT NULL,  -- 'facility' or customer_id
  destination_location_id text NOT NULL,  -- customer_id
  distance_meters numeric NOT NULL,
  duration_seconds numeric NOT NULL,
  instructions jsonb,  -- Array of turn-by-turn instructions
  geometry jsonb,  -- Route path coordinates
  summary jsonb,  -- Route summary metadata
  last_calculated_at timestamptz DEFAULT now(),
  UNIQUE(origin_location_id, destination_location_id)
);
```

### delivery_routes (if multi-stop)

```sql
CREATE TABLE delivery_routes (
  id uuid PRIMARY KEY,
  route_date date NOT NULL,
  driver_id uuid REFERENCES delivery_drivers(id),
  vehicle_id uuid REFERENCES delivery_vehicles(id),
  status text DEFAULT 'planned',  -- 'planned', 'in_progress', 'completed'
  total_distance_meters numeric,
  total_duration_seconds numeric,
  stops_order jsonb,  -- Array of stop IDs in optimized order
  created_at timestamptz DEFAULT now()
);
```
